# JavaScript常用的Hook脚本

# Cookie Hook

var code = function(){
    var org = document.cookie.__lookupSetter__('cookie');
    document.__defineSetter__("cookie",function(cookie){
        // 在这里设置你要hook的cookie名字
        if(cookie.indexOf('cookie_name')>-1){
            debugger;
        }
        org = cookie;
    });
    document.__defineGetter__("cookie",function(){return org;});
}
var script = document.createElement('script');
script.textContent = '(' + code + ')()';
(document.head||document.documentElement).appendChild(script);
script.parentNode.removeChild(script);

// 当cookie中匹配到了 TSdc75a61a， 则插入断点。

# Debugger (构造器方式)

var _constructor = constructor;
Function.prototype.constructor = function(s) {
    if (s == "debugger") {
        console.log(s);
        return null;
    }
    return _constructor(s);
}

# 设置 Window 属性时

// 定义要 hook 属性
var window_flag_1 = "i";   
var window_value = window[window_flag_1];

// hook
Object.defineProperty(window, window_flag_1, {
    get: function(){
        console.log("Getting",window,window_flag_1,"=",window_value);
        //debugger
        return window_value
    },
    set: function(val) {
        console.log("Setting",window, window_flag_1, "=",val);
        debugger
        window_value = val;
    },
})();

# hook请求表单 JSON.stringify()

var my_stringfy = JSON.stringify;
JSON.stringify = function(params) {
    //这里可以添加其他逻辑比如
    debugger
    console.log("json_stringify params:", params);
    return my_stringfy(params);
}
